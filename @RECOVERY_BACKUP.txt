                className={['flex-1 rounded-t-sm transition-all duration-150', val > 75 ? 'bg-indigo-400 shadow-[0_0_8px_#818cf8]' : 'bg-indigo-900/50'].join(' ')} 
                style={{ height: `${Math.min(val, 100)}%` }} 
              />
           ))}
         </div>
      </div>
    </div>
  )
}

function DataConduit({ active, color = 'cyan', pulse = false }: { active: boolean; color?: 'cyan' | 'rose' | 'emerald'; pulse?: boolean }) {
  const colorMap = {
    cyan: 'bg-cyan-400 shadow-[0_0_10px_#22d3ee]',
    rose: 'bg-rose-500 shadow-[0_0_10px_#f43f5e]',
    emerald: 'bg-emerald-500 shadow-[0_0_10px_#10b981]',
  }
  return (
    <div className={['relative w-8 h-[2px] rounded-full mx-1 shrink-0 overflow-hidden transition-colors duration-300', active ? 'bg-slate-700' : 'bg-slate-800'].join(' ')}>
      {active && (
        <div className={`absolute top-0 h-full w-4 ${colorMap[color]} animate-conduit-flow rounded-full`} />
      )}
      {pulse && (
        <div className={`absolute top-0 left-1/2 -translate-x-1/2 h-full w-full ${colorMap[color]} animate-ping opacity-50 rounded-full`} />
      )}
    </div>
  )
}
